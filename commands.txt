# MKDIR #

gcc -o mkdir mkdir.c
./mkdir -p /path/to/new/directory
./mkdir -r /path/to/new/directory

# LS # 

gcc -o ls ls.c
./ls -l
./ls -a
./ls -l -a /path/to/directory

# CP # 

gcc -o cp cp.c
./cp -i source_file destination_folder ( copiar file para um diretório )
./cp -r source_directory destination_directory

# MV # (Move ou renomeia arquivos ou diretório)

gcc -o mv mv.c
./mv source_file destination_file

# GZIP # 

gcc -o gzip gzip.c -lz
./gzip -c original.txt compressed.gz
./gzip -d compressed.gz decompressed.txt
./gzip -z compressed.gz

# DU # 

gcc -o du du.c
./du -h /path/to/directory
./du -s /path/to/directory
./du -a /path/to/directory | sort -n


# PS # 

gcc -o ps_improved ps_improved.c
./ps_improved aux
./ps_improved -u username

# TOP # 

gcc -o custom_top custom_top.c
./custom_top

# FIND # 

gcc -o find_util find_util.c
./find_util -name "pattern" /path/to/search
./find_util -locate "pattern"
./find_util -grep "pattern" /path/to/search
./find_util -fdupes /path/to/search

# TAR #

gcc -o tar_util tar_util.c
./tar_util -c archive.tar file1 file2
./tar_util -x archive.tar
